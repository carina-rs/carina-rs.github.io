Carina

A strongly typed infrastructure management tool written in Rust.

Experimental Carina is under active development. APIs and DSL syntax may change without notice. Not recommended for production use yet.
main.crn 
provider aws {
  region = aws.Region.ap_northeast_1
}

let vpc = awscc.ec2.vpc {
  cidr_block = '10.0.0.0/16'
  enable_dns_support = true
}

awscc.ec2.subnet {
  vpc_id = vpc.vpc_id
  cidr_block = '10.0.1.0/24'
}
terminal
